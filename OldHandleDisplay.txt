static void HandleDisplay() {
        // Implementing Show Production / Consumption
        Console.Clear();

        float AlphaDifference = pending.Alpha;
        float BetaDifference = pending.Beta;
        float GammaDifference = pending.Gamma;
        float EssenceDifference = pending.Essence;

        ConsoleColor BetaProdColor;
        ConsoleColor GammaProdColor;
        ConsoleColor EssenceProdColor;

        if (AlphaDifference > 0) {
                AlphaProdColor = ConsoleColor.Green;
        } else if (AlphaDifference < 0) {
                AlphaProdColor = ConsoleColor.Red;
        } else {
                AlphaProdColor = ConsoleColor.Gray;
        }

        if (BetaDifference > 0) {
                BetaProdColor = ConsoleColor.Green;
        } else if (BetaDifference < 0) {
                BetaProdColor = ConsoleColor.Red;
        } else {
                BetaProdColor = ConsoleColor.Gray;
        }

                GammaProdColor = ConsoleColor.Green;
        } else if (GammaDifference < 0) {
                GammaProdColor = ConsoleColor.Red;
        } else {
                GammaProdColor = ConsoleColor.Gray;
        }

        if (EssenceDifference > 0) {
                EssenceProdColor = ConsoleColor.Green;
        } else if (EssenceDifference < 0) {
                EssenceProdColor = ConsoleColor.Red;
        } else {
                EssenceProdColor = ConsoleColor.Gray;
        }

                Console.WriteLine("");

                Console.WriteLine("");

        }
